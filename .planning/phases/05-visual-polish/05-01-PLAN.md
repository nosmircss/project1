    export interface Particle {

      vx: number        // horizontal velocity
      vy: number        // vertical velocity
      size: number      // radius or length

    `createParticle`: Spawns at random x, y=0 (rain/drizzle/snow fall from top), random x/y for ambient, left edge for fog layers.

    Create `src/renderer/src/components/WeatherParticles.tsx` -- a self-contained React component that renders a `<canvas>` element and drives the particle animation loop.

    **Props interface:**
    ```typescript
    interface WeatherParticlesProps {
      weatherCode: number
      isDay: boolean
      active: boolean  // false during location switch crossfade -- pauses and clears particles per D-11
    }
    ```

    **Component structure:**
    1. `useRef<HTMLCanvasElement>(null)` for canvas DOM reference
    2. `useRef<number>(0)` for RAF handle (for cancellation)
    3. `useRef<Particle[]>([])` for particle pool (mutated in-place, never triggers React re-render)
    4. `useRef<number>(0)` for lightning flash intensity (thunder effect only, per D-08)

    **Main useEffect (dependencies: [weatherCode, isDay, active]):**
    - Get canvas and 2D context from ref. Return early if null.
    - **Canvas sizing (per RESEARCH.md Pitfall 2):**
      ```typescript
      const resize = (): void => {
        const dpr = window.devicePixelRatio || 1
        canvas.width = canvas.offsetWidth * dpr
        canvas.height = canvas.offsetHeight * dpr
        ctx.scale(dpr, dpr)
      }
      ```
      Use `ResizeObserver` on the canvas element. Call `resize()` immediately, then observe.
    - If `active === false`: clear canvas, do NOT start RAF loop, return cleanup that disconnects observer.
    - Get effect config: `const config = getEffectConfig(weatherCode, isDay)`
    - Initialize particle pool: create `config.particleCount` particles using `createParticle(config.effect, canvas.offsetWidth, canvas.offsetHeight)`
    - **RAF tick function:**
      ```typescript
      const tick = (): void => {
        const w = canvas.offsetWidth
        const h = canvas.offsetHeight
        ctx.clearRect(0, 0, w, h)

        // Set glow once per batch (not per particle -- per RESEARCH.md Pitfall 6)
        ctx.shadowColor = config.color
        ctx.shadowBlur = config.glowSize

        // Update and draw all particles
        for (const p of particlesRef.current) {
          updateParticle(p, config.effect, w, h)
          drawParticle(ctx, p, config.effect, config)
        }

        // Reset shadow after batch
        ctx.shadowBlur = 0
        ctx.globalAlpha = 1.0

        // Lightning flash for thunder effect (per D-08)
        if (config.effect === 'thunder' && flashRef.current > 0) {
          ctx.save()
          ctx.fillStyle = `rgba(200, 255, 255, ${flashRef.current * 0.15})`
          ctx.fillRect(0, 0, w, h)
          ctx.restore()
          flashRef.current -= 0.08  // decay per frame
        }

        // Random lightning trigger (~every 1.8s average at 60fps)
        if (config.effect === 'thunder' && Math.random() < 0.0005) {
          flashRef.current = 1.0
        }

        rafRef.current = requestAnimationFrame(tick)
      }
      ```
    - Start loop: `rafRef.current = requestAnimationFrame(tick)`
    - **Cleanup (CRITICAL per STATE.md concern and RESEARCH.md Pitfall 1):**
      ```typescript
      return () => {
        cancelAnimationFrame(rafRef.current)
        rafRef.current = 0
        observer.disconnect()
      }
      ```

    **JSX return:**
    ```tsx
    <canvas
      ref={canvasRef}
      className="absolute inset-0 w-full h-full pointer-events-none"
      style={{ opacity: 0.2, zIndex: 10 }}
    />
    ```
    - `pointer-events: none` per D-06 so clicks pass through to weather content below
    - `opacity: 0.2` per D-06 (~15-25% range, 20% is a good starting point)
    - `zIndex: 10` ensures canvas is above content but below modals
    - `absolute inset-0` positions over the parent (which must have `relative` -- added in Plan 02)
